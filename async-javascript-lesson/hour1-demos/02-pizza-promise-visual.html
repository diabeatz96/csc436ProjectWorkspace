// Creating a Pizza Promise
function orderPizza(type, difficulty) {
    return new Promise((resolve, reject) => {
        console.log(`üìû Order received: ${type} pizza`);
        
        const cookingTime = difficulty === "hard" ? 3000 : 2000;
        const successRate = difficulty === "hard" ? 0.7 : 0.8;
        
        setTimeout(() => {
            if (Math.random() < successRate) {
                resolve(`üéâ ${type} pizza ready!`);
            } else {
                reject(`üòû ${type} ingredients unavailable`);
            }
        }, cookingTime);
    });
}

// Using the Promise
