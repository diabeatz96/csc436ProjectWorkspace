  • Longer total time

    .then(ingredients => {
        console.log("‚úÖ", ingredients);
        return cookPatty();  // Return next promise

        console.log("‚úÖ", patty);
        return toastBuns();  // Chain continues

        console.log("‚úÖ", buns);
        return assembleBurger();

        console.log("üçî Burger ready!");

// Parallel with Promise.all

    getIngredients(),
    cookPatty(),
    toastBuns()

    console.log("All done simultaneously!");
    return assembleBurger();
