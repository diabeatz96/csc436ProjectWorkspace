😵 The Old Way: Promise Chains
📄 promise-chains.js 
// 😵 Promise Chains - Nested and confusing
function makeBurgerOldWay() {
    console.log("🍔 Starting burger...");
    
    getIngredients()
        .then(ingredients => {
            console.log("✅", ingredients);
            return cookPatty();
        })
        .then(patty => {
            console.log("✅", patty);
            return assembleBurger();
        })
        .then(burger => {
            console.log("✅", burger);
            console.log("🏆 Old way complete!");
        })
        .catch(error => {
            console.log("❌ Error:", error);
        });
}
✨ The New Way: Async/Await Magic
✨ async-await-magic.js 
// ✨ Async/Await - Clean and readable
async function makeBurgerNewWay() {
    try {
        console.log("🍔 Starting burger...");
        
        const ingredients = await getIngredients();
        console.log("✅", ingredients);
        
        const patty = await cookPatty();
        console.log("✅", patty);
        
        const burger = await assembleBurger();
        console.log("✅", burger);
        
        console.log("🏆 New way complete!");
        return burger;
    } catch (error) {
        console.log("❌ Error:", error);
    }
}

🔑 Key Async/Await Rules 
// ✅ Key Rules for Async/Await Magic:

// 1. Functions must be marked as 'async'
async function myAsyncFunction() {
    // Can use await inside here
}

// 2. 'await' only works inside async functions
const result = await somePromise(); // ✅ Inside async function
const result = await somePromise(); // ❌ Outside async function

// 3. Async functions always return Promises
async function getName() {
    return "Tony"; // Actually returns Promise.resolve("Tony")
}

// 4. Use try/catch for error handling
try {
    const data = await fetchData();
    // Handle success
} catch (error) {
    // Handle errors
}
