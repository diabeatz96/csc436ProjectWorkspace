üíª The Pokemon Fetch Function

üîç pokemon-fetch.js 
// üîç Complete Pokemon Fetch Function
async function fetchPokemon(pokemonName) {
    try {
        // Step 1: Make the API request
        console.log(`üîç Searching for ${pokemonName}...`);
        const response = await fetch(
            `https://pokeapi.co/api/v2/pokemon/${pokemonName.toLowerCase()}`
        );
        
        // Step 2: Check if request was successful
        if (!response.ok) {
            throw new Error(`Pokemon not found! Status: ${response.status}`);
        }
        
        // Step 3: Parse the JSON response
        console.log("üì¶ Parsing JSON data...");
        const pokemon = await response.json();
        
        // Step 4: Extract useful information
        const pokemonInfo = {
            name: pokemon.name,
            id: pokemon.id,
            height: pokemon.height,
            weight: pokemon.weight,
            types: pokemon.types.map(type => type.type.name),
            abilities: pokemon.abilities.map(ability => ability.ability.name),
            sprite: pokemon.sprites.front_default,
            stats: pokemon.stats.map(stat => ({
                name: stat.stat.name,
                value: stat.base_stat
            }))
        };
        
        return pokemonInfo;
        
    } catch (error) {
        console.log("‚ùå Error catching Pokemon:", error.message);
        return null;
    }
}

// üéÆ Usage Example
fetchPokemon("pikachu").then(pokemon => {
    if (pokemon) {
        console.log("üéâ Pokemon caught!", pokemon);
    } else {
        console.log("üòû Pokemon escaped!");
    }
});

üîç How We Extract the Data

The JSON response has nested objects and arrays. Here's how we extract the information we need:

‚ö° Data Extraction Examples 
// After fetch and response.json(), we have the pokemon object
const pokemon = await response.json();

// ‚úÖ Simple values - direct access
const pokemonName = pokemon.name;           // "pikachu"
const pokemonId = pokemon.id;              // 25
const height = pokemon.height;             // 4
const weight = pokemon.weight;             // 60

// üîç Nested objects - dot notation
const frontSprite = pokemon.sprites.front_default;
// "https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/25.png"

const shinySprite = pokemon.sprites.front_shiny;
// "https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/shiny/25.png"

// üìä Arrays with nested objects - use map() to extract
const types = pokemon.types.map(typeObj => typeObj.type.name);
// Result: ["electric"]

const abilities = pokemon.abilities.map(abilityObj => abilityObj.ability.name);  
// Result: ["static", "lightning-rod"]

const stats = pokemon.stats.map(statObj => ({
    name: statObj.stat.name,
    value: statObj.base_stat
}));
// Result: [{name: "hp", value: 35}, {name: "attack", value: 55}, ...]

// üéØ Why we use map()?
// Arrays need to be processed to extract the data we want
// pokemon.types = [{ slot: 1, type: { name: "electric", url: "..." } }]
// We only want the names: ["electric"]

// Without map() - we get the full objects with extra data
console.log(pokemon.types);
// [{ slot: 1, type: { name: "electric", url: "https://..." } }]

// With map() - we get just the names we need
console.log(pokemon.types.map(t => t.type.name)); 
// ["electric"]

üß† Key JSON Extraction Patterns

üéØ Direct Access: pokemon.name - for simple values

üîç Nested Access: pokemon.sprites.front_default - for nested objects

üìö Array Processing: pokemon.types.map(t => t.type.name) - for arrays
