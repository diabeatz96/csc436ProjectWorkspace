‚è∞ Sequential Fetching

üîÑ

One API at a time

  • Waits for each API to complete
  • Predictable order
  • Longer total time
  • Lower server load

Time = API1 + API2 + API3...

‚ö° Parallel Fetching

üöÄ

All APIs simultaneously

  • Fetches all at once
  • Much faster completion
  • Time = slowest API
  • Better user experience

Time = MAX(API1, API2, API3...)

// ‚è∞ Sequential - One API at a time
async function fetchSequential() {
    try {
        // Each await blocks until complete
        const quote = await getQuote();
        displayQuote(quote);
        
        const joke = await getJoke();
        displayJoke(joke);
        
        const fact = await getFact();
        displayFact(fact);
        
        const advice = await getAdvice();
        displayAdvice(advice);
        
    } catch (error) {
        console.error("Sequential fetch failed:", error);
    }
}
‚ö° parallel-fetch.js 
// ‚ö° Parallel - All APIs at once
async function fetchParallel() {
    try {
        // Start all requests simultaneously
        const [quote, joke, fact, advice] = await Promise.all([
            getQuote(),
            getJoke(),
            getFact(),
            getAdvice()
        ]);
        
        // All data arrives together!
        displayQuote(quote);
        displayJoke(joke);
        displayFact(fact);
        displayAdvice(advice);
        
    } catch (error) {
        console.error("Parallel fetch failed:", error);
    }
}

// üî• Alternative: Handle each result as it arrives
async function fetchWithSettled() {
    const results = await Promise.allSettled([
        getQuote(),
        getJoke(),
        getFact(),
        getAdvice()
    ]);
    
    results.forEach(result => {
        if (result.status === 'fulfilled') {
            console.log('Success:', result.value);
        } else {
            console.log('Failed:', result.reason);
        }
    });
}
